14 // These functions are used to dynamically allocate and free shared_ptr
15 // objects. shared_ptr is just an alias for std::shared_ptr, which does not
16 // derive from IAllocatable. Therefore, it cannot use a custom allocator that
17 // the user may set using iplSetMemoryAllocator. In order to make this work,
18 // we allocate shared_ptr objects using Memory::allocate (effectively
19 // malloc, unless replaced) and then use placement new to call the constructor.
20 // Similarly, when we want to destroy a shared_ptr, we explicitly call
21 // the destructor, then use Memory::free (effectively free, unless replaced)
22 // to free the memory.
23 
24 template <typename T> Handle<T> createHandle(shared_ptr<T> sharedPointer)
25 {
26  Handle<T> handle = reinterpret_cast<Handle<T>>(Memory::allocate(sizeof(shared_ptr<T>), IPL_ALIGNMENT_SIMD));
27  new (handle) shared_ptr<T>(sharedPointer);
28  return handle;
29 }
30 
31 template <typename T> void destroyHandle(Handle<T> handle)
32 {
33  handle->~shared_ptr();
34  Memory::free(handle);
35 }

38 inline AudioFormat reinterpretAudioFormat(IPLAudioFormat audioFormat)
39 {
40  if (audioFormat.channelLayoutType == IPL_CHANNELLAYOUTTYPE_AMBISONICS)
41  {
42  return AudioFormat{ audioFormat.ambisonicsOrder,
43  static_cast<AmbisonicsOrdering>(audioFormat.ambisonicsOrdering),
44  static_cast<AmbisonicsNormalization>(audioFormat.ambisonicsNormalization),
45  static_cast<ChannelOrder>(audioFormat.channelOrder) };
46  }
47  else if (audioFormat.channelLayout == IPL_CHANNELLAYOUT_CUSTOM)
48  {
49  return AudioFormat{ audioFormat.numSpeakers, reinterpret_cast<Vector3f*>(audioFormat.speakerDirections),
50  static_cast<ChannelOrder>(audioFormat.channelOrder) };
51  }
52  else
53  {
54  return AudioFormat{ static_cast<ChannelLayout>(audioFormat.channelLayout),
55  static_cast<ChannelOrder>(audioFormat.channelOrder) };
56  }
57 }
